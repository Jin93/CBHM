model
{
  for (i in 1:K)
  {
    Y[i] ~ dbin(p[i],n[i])
    p[i] <- exp(theta[i])/(1+exp(theta[i]))
    theta[i] <- theta0 + e[i]
    e[i] ~ dnorm(0,tausq)
  }
  theta0 ~ dnorm(-mu0,0.001)
  tausq ~ dgamma(0.001,0.001)
}


#### mu0 = log(q0/(1-q0)) where q0 denotes the standard of care response rate
